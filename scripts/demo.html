calculateDaysBetweenDates()

// Calculate days between two dates
const days = calculateDaysBetweenDates('2024-01-01', '2024-01-10');
console.log(days); // Output: 9

// Using Date objects
const days2 = calculateDaysBetweenDates(
    new Date('2024-01-01'),
    new Date('2024-01-10')
);
console.log(days2); // Output: 9
